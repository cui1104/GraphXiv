    ```python
    """Paired A/B agent runner for Phase 8 evaluation (EVAL-02).

    Executes every question from eval/questions.json under two conditions:
      - baseline: agent tool list excludes fetch_cited_paper_sections via a module-level
        monkey-patch of deepxiv_sdk.agent.graph.get_tools_definition (D-15 corrected).
      - citation_aware: Agent(citation_depth=1), full tool list (D-09, D-10).

    Writes resumable JSONL to eval/results/runs.jsonl per D-13 / D-14.
    Schema per row (D-13, this file):
      { jsonl_schema_version, run_id, question_id, condition, model,
        timestamp, prompt_hash, system_fingerprint, answer_text,
        tool_calls, tokens_used, latency_s, error }
    """

    import argparse
    import contextlib
    import hashlib
    import json
    import logging
    import os
    import sys
    import time
    import uuid
    from datetime import datetime, timezone
    from pathlib import Path
    from typing import Callable, Iterable, Optional
    from unittest.mock import patch

    sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

    JSONL_SCHEMA_VERSION = 1
    DEFAULT_QUESTIONS_PATH = Path(__file__).parent / "questions.json"
    DEFAULT_RESULTS_PATH = Path(__file__).parent / "results" / "runs.jsonl"
    DEFAULT_MODEL = "gpt-4o"               # D-10
    DEFAULT_BASE_URL = "http://localhost:8000"
    EXCLUDED_BASELINE_TOOL = "fetch_cited_paper_sections"  # D-08: only this tool returns cited-paper bodies

    logger = logging.getLogger("run_eval")
    logging.basicConfig(level=logging.INFO, format="%(asctime)s %(levelname)s %(message)s")


    # ---------- Tool-subset isolation (D-15 corrected per 08-RESEARCH.md Pattern 2) ----------
    #
    # The langgraph-based Agent imports get_tools_definition at MODULE LEVEL inside
    # deepxiv_sdk/agent/graph.py. Patching ToolExecutor.get_tools_definition (instance
    # method) will NOT affect graph construction. Patch the module symbol.

    def build_baseline_tools_fn(original_fn: Callable[..., list[dict]]) -> Callable[..., list[dict]]:
        """Wrap get_tools_definition so the returned tool list excludes EXCLUDED_BASELINE_TOOL."""
        def _filtered(*args, **kwargs) -> list[dict]:
            tools = original_fn(*args, **kwargs) or []
            return [t for t in tools if (t.get("function", {}).get("name") or t.get("name")) != EXCLUDED_BASELINE_TOOL]
        _filtered.__wrapped__ = original_fn  # for test introspection
        return _filtered


    @contextlib.contextmanager
    def _baseline_tool_isolation():
        """Patch deepxiv_sdk.agent.graph.get_tools_definition to drop fetch_cited_paper_sections."""
        # Import target via a late import so tests can import this module without side effects.
        import deepxiv_sdk.agent.graph as graph_mod
        original = graph_mod.get_tools_definition
        filtered = build_baseline_tools_fn(original)
        with patch.object(graph_mod, "get_tools_definition", filtered):
            yield
        # patch.object restores on exit.


    # ---------- JSONL I/O (08-RESEARCH.md Pattern 3) ----------

    def load_done_pairs(results_path: Path, skip_errors: bool = True) -> set[tuple[str, str]]:
        """Return the set of (question_id, condition) pairs already completed successfully.

        When skip_errors=True (default): rows with error != None are NOT counted as done
        so the runner re-tries them on next run (D-14: resumable with error-retry opt-in).
        """
        path = Path(results_path)
        if not path.exists():
            return set()
        done: set[tuple[str, str]] = set()
        with open(path, "r", encoding="utf-8") as f:
            for line in f:
                line = line.strip()
                if not line:
                    continue
                try:
                    row = json.loads(line)
                except json.JSONDecodeError:
                    continue
                if skip_errors and row.get("error"):
                    continue
                qid = row.get("question_id")
                cond = row.get("condition")
                if qid and cond:
                    done.add((qid, cond))
        return done


    def append_row(results_path: Path, row: dict) -> None:
        path = Path(results_path)
        path.parent.mkdir(parents=True, exist_ok=True)
        # Append-only; one JSON object per line per D-13.
        with open(path, "a", encoding="utf-8") as f:
            f.write(json.dumps(row, ensure_ascii=False) + "\n")
            f.flush()


    # ---------- Row construction ----------

    def _prompt_hash(system_prompt: str, question_text: str) -> str:
        h = hashlib.sha256()
        h.update(system_prompt.encode("utf-8"))
        h.update(b"\n---\n")
        h.update(question_text.encode("utf-8"))
        return h.hexdigest()[:16]


    def _now_iso() -> str:
        return datetime.now(timezone.utc).isoformat(timespec="seconds").replace("+00:00", "Z")


    def _extract_tool_calls(agent_result: dict) -> list[dict]:
        """Normalize the agent_result['tool_calls'] shape so downstream scoring code is simple.

        Returns list of {name, arguments, arxiv_id_hit} (the last is the arxiv_id that this
        tool call acted on, pulled from 'arguments' when present — used by the deterministic
        citation_coverage check in 08-03 Pattern 5).
        """
        out: list[dict] = []
        for tc in (agent_result or {}).get("tool_calls", []) or []:
            name = tc.get("name") or tc.get("function", {}).get("name")
            args = tc.get("arguments") or tc.get("function", {}).get("arguments")
            if isinstance(args, str):
                try:
                    args = json.loads(args)
                except json.JSONDecodeError:
                    args = {"__raw__": args}
            arxiv_hit = None
            if isinstance(args, dict):
                for k in ("arxiv_id", "paper_id", "id"):
                    if args.get(k):
                        arxiv_hit = args[k]
                        break
            out.append({"name": name, "arguments": args, "arxiv_id_hit": arxiv_hit})
        return out


    def _build_row(
        *,
        question: dict,
        condition: str,
        model: str,
        agent_result: dict,
        system_prompt: str,
        latency_s: float,
        error: Optional[str],
    ) -> dict:
        return {
            "jsonl_schema_version": JSONL_SCHEMA_VERSION,
            "run_id": str(uuid.uuid4()),
            "question_id": question["question_id"],
            "condition": condition,
            "model": model,
            "timestamp": _now_iso(),
            "prompt_hash": _prompt_hash(system_prompt, question["question_text"]),
            "system_fingerprint": (agent_result or {}).get("system_fingerprint"),
            "answer_text": (agent_result or {}).get("answer") if not error else None,
            "tool_calls": _extract_tool_calls(agent_result) if not error else [],
            "tokens_used": (agent_result or {}).get("usage", {}).get("total_tokens") if not error else None,
            "latency_s": round(latency_s, 3),
            "error": error,
        }


    # ---------- Per-(question, condition) run ----------

    def run_one(
        question: dict,
        condition: str,
        *,
        base_url: str = DEFAULT_BASE_URL,
        model: str = DEFAULT_MODEL,
    ) -> dict:
        """Run a single Agent call under the given condition. Returns a D-13 row.

        Baseline: patch deepxiv_sdk.agent.graph.get_tools_definition (D-15 corrected).
        Citation-aware: Agent(citation_depth=1), full tools (D-09).
        """
        from deepxiv_sdk.agent import Agent  # lazy to keep import cheap in tests
        citation_depth = 0 if condition == "baseline" else 1
        system_prompt = "You are a research assistant that answers questions about arXiv papers using the provided tools."

        t0 = time.time()
        try:
            if condition == "baseline":
                with _baseline_tool_isolation():
                    agent = Agent(base_url=base_url, model=model, citation_depth=citation_depth, reset_papers=True)
                    result = agent.run(question["question_text"])
            else:
                agent = Agent(base_url=base_url, model=model, citation_depth=citation_depth, reset_papers=True)
                result = agent.run(question["question_text"])
            error = None
        except Exception as exc:
            result = None
            error = f"{type(exc).__name__}: {exc}"
            logger.warning("run_one(%s,%s) errored: %s", question["question_id"], condition, error)
        latency_s = time.time() - t0

        return _build_row(
            question=question,
            condition=condition,
            model=model,
            agent_result=result or {},
            system_prompt=system_prompt,
            latency_s=latency_s,
            error=error,
        )


    # ---------- Main loop ----------

    def run_all(
        questions_path: Path = DEFAULT_QUESTIONS_PATH,
        results_path: Path = DEFAULT_RESULTS_PATH,
        *,
        base_url: str = DEFAULT_BASE_URL,
        model: str = DEFAULT_MODEL,
        conditions: Iterable[str] = ("baseline", "citation_aware"),
        retry_errors: bool = False,
        limit: Optional[int] = None,
        dry_run: bool = False,
    ) -> dict:
        with open(questions_path) as f:
            doc = json.load(f)
        questions = doc["questions"]
        done = load_done_pairs(results_path, skip_errors=not retry_errors)

        pending: list[tuple[dict, str]] = []
        for q in questions:
            for cond in conditions:
                if (q["question_id"], cond) in done:
                    continue
                pending.append((q, cond))

        if limit is not None:
            pending = pending[:limit]

        logger.info("planned: %d pairs (%d already done, %d to execute)", len(pending) + len(done), len(done), len(pending))
        if dry_run:
            return {"planned": len(pending), "already_done": len(done)}

        for q, cond in pending:
            row = run_one(q, cond, base_url=base_url, model=model)
            append_row(results_path, row)
            logger.info("wrote run qid=%s cond=%s err=%s", q["question_id"], cond, row["error"])
        return {"written": len(pending), "already_done": len(done)}


    # ---------- CLI ----------

    def main() -> int:
        parser = argparse.ArgumentParser(description="Phase 8 paired A/B agent runner (EVAL-02).")
        parser.add_argument("--questions-path", type=Path, default=DEFAULT_QUESTIONS_PATH)
        parser.add_argument("--results-path", type=Path, default=DEFAULT_RESULTS_PATH)
        parser.add_argument("--base-url", type=str, default=DEFAULT_BASE_URL)
        parser.add_argument("--model", type=str, default=DEFAULT_MODEL)
        parser.add_argument("--conditions", nargs="+", default=["baseline", "citation_aware"])
        parser.add_argument("--retry-errors", action="store_true", help="Re-run rows with error != None on resume")
        parser.add_argument("--limit", type=int, default=None)
        parser.add_argument("--dry-run", action="store_true")
        args = parser.parse_args()
        summary = run_all(
            questions_path=args.questions_path,
            results_path=args.results_path,
            base_url=args.base_url,
            model=args.model,
            conditions=tuple(args.conditions),
            retry_errors=args.retry_errors,
            limit=args.limit,
            dry_run=args.dry_run,
        )
        print(f"[run_eval] {summary}")
        return 0


    if __name__ == "__main__":
        sys.exit(main())
    ```
